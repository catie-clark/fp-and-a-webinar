      switch (field) {
        case 'revenueGrowthPct':
          return `${v >= 0 ? '+' : ''}${(v * 100).toFixed(1)}%`;
        case 'grossMarginPct':
        case 'collectionsRatePct':
        case 'returnsPct':
          return `${(v * 100).toFixed(1)}%`;
        case 'fuelIndex':
          return `${Math.round(v)}`;
        case 'lateInvoiceHours':
          return `${v % 1 === 0 ? v.toFixed(0) : v.toFixed(1)} hrs`;
        case 'journalLoadMultiplier':
          return `${v.toFixed(2)}x`;
        default:
          return String(v);
      }
    };
